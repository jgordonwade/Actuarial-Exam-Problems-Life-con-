\JGWitem{May 2000, \#3}
 An insurance on (x) pays 400 at the end of the year of death if (x) dies within the next two years. 
\begin{enumerate}
\item The annual premium amount $P$  for this policy is the net level annual premium, which is \$74.33.
\item  The effective annual interest rate is 10\%.
\item The policy value $_1\!V$ after one year is \$16.58
\end{enumerate}

\bigskip

Calculate the variance of the loss at issue.

\showMC{\begin{center}
 (A) 21,615 {\ \ \ \ \ }
 (B) 23,125 {\ \ \ \ \ }
 (C) 27,450 {\ \ \ \ \ }
 (D) 31,175 {\ \ \ \ \ }
 (E) 34,150
\end{center}}

\showsol{\bsoln 
We have:
\bears
  \prepostsub{0}{L}{} = 400\termify{Z_{x:\halfbox{2}}} - P\ddot{Y}_{x:\halfbox{2}} 
   &=& \left\{\begin{array}{ll} 
      400v - P & \mbox{w/ prob. }q_x \\
      400v^2 - P(1+v) & \mbox{w/ prob. }\prepostsub{1|}{q}{x} = (1-\qx{x})\qx{x+1} \\
      - P(1+v) & \mbox{w/ prob. }\prepostsub{2}{p}{x} = (1-\qx{x})(1-\qx{x+1})
      \end{array}\right. \\
   &=& \left\{\begin{array}{ll} 
       289.30 & \mbox{w/ prob. }q_x \\
       188.67 & \mbox{w/ prob. }\prepostsub{1|}{q}{x} \\
      -141.90 & \mbox{w/ prob. }\prepostsub{2}{p}{x}
      \end{array}\right.
\eears
%and since $\Expect{\prepostsub{0}{L}{}}=0$ we have $\Var{\prepostsub{0}{L}{}} = \Expect{\prepostsub{0}{L}{}^2}$, so that
%\be \label{May2000_03_1}
%   \Var{\prepostsub{0}{L}{}} =  q_x(0.723266)^2 + \prepostsub{1|}{q}{x}(0.471689)^2 + \prepostsub{2}{p}{x}(0.354757)^2.
%\ee
So we need the probability values $q_x$ and $q_{x+1}$.

\underline{ To find $q_{x+1}$}, use the given value of $_1\!V$:
\bears
 _1\!V = 16.58 &=& 400\Ax{\term{x+1}{1}} - P\ax**{\endow{x+1}{1}} \\
   &=& 400vq_{x+1} - P = \dfrac{400}{1.1}\qx{x+1} - 74.33 \quad\quad\Longrightarrow \quad  q_{x+1} = 0.2500025
\eears

\underline{To find $q_{x}$}, use the fact that $_0\!V=0$ (because $P$ is a {\em net} premium), and
a recursive formula to relate $_0\!V$ to $_1\!V$:
\bears
   0 \;=\; _0\!V &=& 400vq_{x} - P + \Ex{x}\cdot (_1\!V) \\
   0 &=& \dfrac{400}{1.1} q_x - 74.33 + \dfrac{1-q_x}{400}\cdot 16.58 
   \quad\quad\Longrightarrow \quad q_x = 0.1700041730
\eears

So we have
\bears
   q_x &=& 0.1700041730 \\
   \prepostsub{1|}{q}{x} &=& (1-q_x)q_{x+1} = 0.207501 \\
   \prepostsub{2}{p}{x} &=& 1 - q_x - \prepostsub{1|}{q}{x}  = 0.622494827
\eears
So
\bears
  \prepostsub{0}{L}{} &=& 400\termify{Z_{x:\halfbox{2}}} - P\ddot{Y}_{x:\halfbox{2}} 
   = \left\{\begin{array}{ll} 
      289.30 & \mbox{w/ prob. } 0.1700041730 \\
      188.67 & \mbox{w/ prob. }0.207501 \\
      -141.90 & \mbox{w/ prob. }0.622494827
      \end{array}\right.
\eears
and so $\Var{_0\!L} = 34,149.$


\esoln}
