\JGWitem{Nov 2000, \#14 (book $6^{th}$ ed, \#13.12)}
For a double decrement table with $l_{40}^{(\tau)}= 2000:$
\begin{center}\begin{tabular}{|c|c|c|c|c|}\hline
   $x$  &  $q_x^{(1)}$  &   $q_x^{(2)}$  &   $q_x^{'(1)}$  &  $q_x^{'(2)}$   \\ \hline
   40   &  0.24       &     0.10      &    0.25        & y             \\ \hline
   41   &  ---        &     ---       &    0.20        & 2y             \\ \hline
\end{tabular}\end{center}
Calculate $l_{42}^{(\tau)}$
\showMC{\begin{description}
\item[(A)] 800
\item[(B)] 820
\item[(C)] 840
\item[(D)] 860
\item[(E)] 880
\end{description}}
\showsol{\bsoln
 \underline{ANS: (A)}
 \bears
   q_{40}^{(\tau)} &=& q_{40}^{(1)} + q_{40}^{(2)} = 0.24 + 0.10 = 0.34, \\
  \mbox{and also }   q_x^{(\tau)} &=& 1-p_x^{(\tau)} = 1-(p_{40}^{'(1)})(p_{40}^{'(2)}) = 1-(1-0.25)(1-y) \\
  \mbox{so, } 0.34 &=& 1-(1-0.25)(1-y) \\
  && \Longrightarrow \underline{ y = 0.12} \\
  p_{40}^{(\tau)} &=& 1 - q_{40}^{(\tau)} = 1-0.34 = 0.66\\
  p_{41}^{(\tau)} &=& (1-q_{41}^{'(1)})(1-q_{40}^{'(2)}) = (1-0.20)(1-2y) = (0.8)(0.76) = 0.608 \\
  \prepostsubsup{2}{p}{40}{(\tau)} &=& (p_{40}^{(\tau)})(p_{41}^{(\tau)}) = (0.66)(0.608) = 0.40128 \\
  l_{42} &=& (\prepostsubsup{2}{p}{40}{(\tau)})\cdot l_{40} = 2000\cdot(0.40128) = 802.56.
\eears
\esoln}
