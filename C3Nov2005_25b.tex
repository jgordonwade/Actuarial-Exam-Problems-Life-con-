\JGWitem{ C3, Nov 2005, \#25b}
An insurance on (x) pays $300,000+50,000K_{x}$ if $K_{x}\leq 2$ and zero otherwise. 
Given that $\qx{x+k} = 0.02(k+1)$ for $k=0,1,2$ and the $i=0.06,$ compute the mean and variance of 
the present value random variable for this insurance.
\showsol{\bsoln
\begin{center}\begin{tabular}{|c|c|c|} \hline 
   $\prepostsub{0|}{q}{x}=\Prob{K_x=0}$ & $\prepostsub{1|}{q}{x}=\Prob{K_x=1}$ & $\prepostsub{2|}{q}{x}=\Prob{K_x=2}$ \\ \hline && \\ 
   \underline{$0.02$}                                & $0.98\cdot 0.04=\underline{0.0392}$ &  $0.98\cdot 0.96 \cdot 0.06 = \underline{0.056448}$  \\ && \\ \hline
\end{tabular}\end{center}


\bears
   Z &=& \left\{\begin{array}{cl} 
                300\cdot 1.06^{-1} &  \mbox{ with probability }0.02, \\
                350\cdot 1.06^{-2} &  \mbox{ with probability }0.0392, \\
                400\cdot 1.06^{-3} &  \mbox{ with probability }0.056448, \\
                 0 & \mbox{otherwise.}
   \end{array}\right. \\ \ \\
   \Expect{Z} &=& 0.02\cdot 300\cdot 1.06^{-1} + 0.0392\cdot 350\cdot 1.06^{-2}  + 0.056448\cdot 400\cdot 1.06^{-3}  \\ &=& 36.829 \\
   \Expect{Z^2} &=& 0.02\cdot (300\cdot 1.06^{-1})^2 + 0.0392\cdot (350\cdot 1.06^{-2})^2  + 0.056448(\cdot 400\cdot 1.06^{-3})^2  \\ &=& 11772.6 \\ 
   && \\
   \Var{Z} &=& \Expect{Z^2}-\Expect{Z}^2 =  11772.6 - (36.829)^2 = \boxed{10416.2}
%
%
\eears
\esoln}
