\JGWitem{Nov 2000, \#36}   % \S2 Single Life Survival
 Given:
\begin{enumerate}
\item $\mu(x) = F +e^{2x}$, $x\geq 0$.
\item $\prepostsub{0.4}{p}{0}=0.5$.
\end{enumerate}
Calculate F.
\showMC{\begin{description}
\item[(A)] -0.20
\item[(B)] -0.09
\item[(C)] 0.00
\item[(D)] 0.09
\item[(E)] 0.20
\end{description}}

\showsol{\bsoln
\bears
   \underline{\frac{1}{2} = \prepostsub{0.4}{p}{0} }&=& S_0(2/5) = \exp\left(-\int_{z=0}^{2/5} \mu(z)\,dz \right)  \\
               &=& \exp\left(-\int_{z=0}^{2/5} F + e^{2z}\,dz\right) \\ \\
                    && \quad\quad \quad \left[-\int_{z=0}^{4/5} F + e^{2z}\, dz = \frac{-2}{5}F + \frac{1}{2}\left(1-e^{4/5}\right)\right] \\ \\
         \frac{1}{2}      &=& \exp\left(\frac{-2}{5}F + \frac{1}{2}\left(1-e^{4/5}\right)\right)
\eears
Now solve for $F$: \fbox{$F=0.200942$}


\esoln}
