\JGWitem{MLC May 2016, \#2}   % \S2 Single Life Survival


You are given the survival function:
\[
   S_0(x) = \left(1-\dfrac{x}{60}\right)^{1/3}, \quad\quad 0\leq x\leq 60.
\]
Calculate $1000\,\mu_{35}$

\showsol{\bsoln
  \bears
       \frac{-d}{dx} S_0(x) &=& \frac{1}{3\cdot 60}\left(1-\dfrac{x}{60}\right)^{-2/3} \\
       \mu_x &=& \dfrac{\dfrac{-d}{dx} S_0(x)}{S_0(x)} = \frac{1}{3\cdot 60}\left(1-\dfrac{x}{60}\right)^{-2/3 - 1/3} = \frac{1}{3\cdot 60\cdot(1-x/60)} \\
       && \boxed{\mu_x = \frac{1}{3\cdot (60-x)}} \\
       \mu_{35} &=& \frac{1}{3\cdot 25} = \frac{1}{75} = \boxed{\underline{0.013333333}}
   \eears
\esoln}
