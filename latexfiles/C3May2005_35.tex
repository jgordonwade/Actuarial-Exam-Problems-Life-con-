\JGWitem{May 2005 \#35}
 For a special fully discrete 5-year deferred whole life insurance of 100,000 on (40), you are
given:
\begin{enumerate}
\item The death benefit during the 5-year deferral period is return of benefit premiums paid
  without interest.
\item Annual benefit premiums are payable only during the deferral period.
\item Mortality follows the Illustrative Life Table.
\item $i = 0.06$
\item $\termify{(IA)_{40:\halfbox{5}}}= 0.04042$
\end{enumerate}
Calculate the annual benefit premiums.
\showMC{\begin{description}
\item[(A)] 3300
\item[(B)] 3320
\item[(C)] 3340
\item[(D)] 3360
\item[(E)] 3380
\end{description}}
\showsol{\bsoln
   APV of Benefits paid: $10^5 \prepostsub{5|}{A}{40} + \pi\termify{(IA)_{40:\halfbox{5}}}$

   APV of Premiums  received: $\pi\ddot{a}_{40:\halfbox{5}}$

   \bears
     \Expect{L} &=& 0 = 10^5 \prepostsub{5|}{A}{40} + \pi\termify{(IA)_{40:\halfbox{5}}} - \pi\ddot{a}_{40:\halfbox{5}} \\
        &=& 10^5 \prepostsub{5|}{A}{40} - \pi\left[\pi\ddot{a}_{40:\halfbox{5}} - \termify{(IA)_{40:\halfbox{5}}}\right] \\
      \pi &=& 10^5 \prepostsub{5|}{A}{40}/\left[\pi\ddot{a}_{40:\halfbox{5}} - \termify{(IA)_{40:\halfbox{5}}}\right]  \\
          &=&  (0.20120)\cdot(0.73529) / \bigl[4.4401 - 0.04042\bigr] = 3363
   \eears

\esoln}
