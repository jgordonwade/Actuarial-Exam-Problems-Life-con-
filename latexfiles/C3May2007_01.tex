\JGWitem{May 2007 \#1}   % \S2 Single Life Survival
You are given: 
\begin{enumerate}
  \item $\prepostsub{3}{p}{70} = 0.95$
  \item $\prepostsub{2}{p}{71} = 0.96$
  \item $\int_{71}^{75} \mu_x\,dx = 0.107$. 
\end{enumerate}
Calculate $\prepostsub{5}{p}{70}.$

\showsol{\bsoln
\underline{Strategy:} Write $\prepostsub{5}{p}{70} = (p_{70})\cdot(\prepostsub{4}{p}{71}).$ \\ Use (a) and (b) to get $p_{70}$, \\ and use (c) to get $\prepostsub{4}{p}{71}.$ 


\bigskip
Use (a) and (b) to get $p_{70}$.
\bears
  \frac{\prepostsub{3}{p}{70}}{\prepostsub{2}{p}{71}} &=& \frac{S(73)/S(70)}{S(73)/S(71)} = \frac{S(71)}{S(70)} = p_{70} \\ \\
  \underline{p_{70}= {0.95}/{0.96} }
\eears  

\smallskip
Use (c) to get $\prepostsub{4}{p}{71}$:
\bears
   \prepostsub{t}{p}{x} = \exp\left(-\int_{z=0}^t \mu(x+z)\,dz\right) &=& \exp\left(-\int_{x}^{x+t} \mu(z)\,dz\right) \\ 
     &&    \exp\left(-\int_{71}^{75} \mu_x(z)\,dz\right) = e^{-0.107} \\
   \underline{\prepostsub{4}{p}{71}    =    e^{-0.107}}.
\eears

\smallskip
Now,
\[ 
   \prepostsub{5}{p}{70} = p_{70}\prepostsub{4}{p}{71} = \frac{95e^{-0.107}}{96} \approx \underline{0.8892 = \prepostsub{5}{p}{70}}.
\]

 \esoln}
