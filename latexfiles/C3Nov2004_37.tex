\JGWitem{Nov 2004, \#37}
Z is the present value random variable for a 15-year pure endowment of 1 on (x):
\begin{enumerate}
\item The force of mortality is constant over the 15-year period.
\item $v = 0.9.$
\item $\Var{Z}=0.065\Expect{Z}$.
\end{enumerate}
Calculate $q_x$.
\showsol{\bsoln
\bears
    Z &=& \left\{\begin{array}{cl} v^{15} & \mbox{ if (x) survives 15 years, }\\ 0 & \mbox{otherwise}.\end{array}\right. \\
    \Prob{\mbox{(x) survives 15 yrs }} &=& \prepostsub{15}{p}{x} = e^{-15 \mu} \quad\mbox{ \ since ``constant force''} \\
    Z &=& \left\{\begin{array}{cl} v^{15} & \mbox{w/ prob. }e^{-15\mu}\\ 0 & \mbox{w/ prob. }1-e^{-15\mu}.\end{array}\right.\\
    \Expect{Z} &=& v^{15}e^{-15\mu} \\
    \Expect{Z^2} &=& (v^{15})^2e^{-15\mu} \\
    \Var{Z} &=&     \Expect{Z^2} -  \Expect{Z}^2 = v^{30}e^{-15\mu}\Bigl(1-e^{-15\mu}\Bigr) \\
    \Var{Z}=0.065\Expect{Z} \;\Longrightarrow\; \frac{65}{1000} &=& \dfrac{v^{30}e^{-15\mu}\Bigl(1-e^{-15\mu}\Bigr)}{v^{15}e^{-15\mu}} = v^{15}(1-e^{-15\mu}) \\
      1-\frac{65}{1000\cdot 0.9^{15}} &=& e^{-15\mu} \\ \\
                                           && \underline{\mu = 0.025290671} \\ \\
                                           q_x &=& 1-e^{-\mu} \\
                                           && \boxed{q_x = 0.02497}
\eears
\esoln}
