From the Equiv. Princ. we have
\bears
    0&=&
    \overbrace{100,000\Ax*{\term{50}{20}}}^{\text{death ben.}}  
     + \overbrace{R\ax*{\endow{50}{20}} }^{\text{expenses}} 
     -  \overbrace{ 4500\ax*{\endow{50}{20}} }^{\text{premiums}} \\
\eears
Using the constant forces, 
\bears
   \Ax*{\term{50}{25}} &=& \Ax*{50} - \Ex[20]{50}\Ax*{70}  =
   \dfrac{\mu}{\mu+\delta} - e^{-20(\delta+\mu)}\dfrac{\mu}{\mu+\delta}  \\ &=& 
   \dfrac{0.04}{0.04+0.08} - e^{-20(0.08+0.04)}\dfrac{0.04}{0.04+0.08} \\ &=& 0.303094 \\ \\ 
   \ax*{\endow{50}{20}}  &=&    \dfrac{1}{\mu+\delta} - e^{-20(\delta+\mu)}\dfrac{1}{\mu+\delta}  \\
%     = \dfrac{1}{\mu}\cdot \left(   \dfrac{\mu}{\mu+\delta} - e^{-20(\delta+\mu)}\dfrac{\mu}{\mu+\delta} \right)
%     = \dfrac{1}{\mu} \Ax*{\term{50}{25}} = 
    &=& = \dfrac{1}{0.04+0.08} - e^{-20(0.08+0.04)}\dfrac{1}{0.04+0.08} \\ &=& 7.57735,
\eears
so that
\bears
    0&=& 100,000\Ax*{\term{50}{20}} + R\ax*{\endow{50}{20}}  -  4500\ax*{\endow{50}{20}} \\
    0 &=& 100,000\cdot 0.303094 - 4500\cdot 7.57735 + R\cdot 7.57735 \\
    && \boxed{R=500}
\eears    


\esoln}
