\JGWitem{May 2000, \#15}
In a double decrement table:
\begin{enumerate}
\item $l_{30}^{(\tau)} = 1000$
\item $q^{'(1)}_{30} = 0.100$
\item $q^{'(2)}_{30} = 0.300$
\item $\prepostsubsup{1|}{q}{30}{(1)} = 0.075$
\item $l_{32}^{(\tau)} = 472$
\end{enumerate}
Calculate $q_{31}^{(2)}$

\showMC{\begin{description}
\item[(A)] 0.11
\item[(B)] 0.13
\item[(C)] 0.14
\item[(D)] 0.15
\item[(E)] 0.17
\end{description}}

\showsol{\bsoln
\showMC{\underline{ANS: (B)}, here how. }First, isolate $q_{31}^{(2)}$ in ``$\prepostsubsup{2}{q}{30}{(\tau)} = q_{30}^{(\tau)} + \prepostsubsup{1|}{q}{30}{(\tau)}$'',
\bears
   \prepostsubsup{2}{q}{30}{(\tau)} &=& q_{30}^{(\tau)} + \prepostsubsup{1|}{q}{30}{(\tau)} \\
                         &=& q_{30}^{(\tau)} + p_{30}^{(\tau)}\,\cdot\,q_{31}^{(\tau)} \\
                         &=& q_{30}^{(\tau)} + p_{30}^{(\tau)}\,\cdot\,\bigl(q_{31}^{(1)} + q_{31}^{(2)} \bigr)\\
   \prepostsubsup{2}{q}{30}{(\tau)} &=& q_{30}^{(\tau)} + \prepostsubsup{1|}{q}{30}{(1)} + p_{30}^{(\tau)}\,\cdot\,\underline{q_{31}^{(2)} } \\
   && \quad\Longrightarrow \quad \underline{q_{31}^{(2)}  = \Bigl(\prepostsubsup{2}{q}{30}{(\tau)} - q_{30}^{(\tau)} - \prepostsubsup{1|}{q}{30}{(1)}\Bigr)/p_{30}^{(\tau)}} 
\eears
Now get the three numbers on the right side of the last equation:
\bears
     \prepostsubsup{2}{q}{30}{(\tau)} &=& 1- \prepostsubsup{}{p}{30}{(\tau)} =  1 - \frac{l_{32}^{(\tau)}}{l_{30}^{(\tau)}} = 1-\frac{472}{1000} = 0.528 \\
     \prepostsubsup{1|}{q}{30}{(\tau)} &=& 0.075 \quad\mbox{(given)}, \\
      p_{30}^{(\tau)} &=& \left(p^{'(1)}_{30}\right)\left(p^{'(2)}_{30}\right) 
                         = \left(1-q^{'(1)}_{30}\right)\left(1-q^{'(2)}_{30}\right) = (1-0.100)(1-0.300) = 0.630 \\
      q_{30}^{(\tau)} &=& 1-      p_{30}^{(\tau)} = 0.37, 
\eears
So: \[ \underline{q_{31}^{(2)} = \frac{0.528-0.37-0.075}{0.630} = 0.1317.}\]
\esoln}
