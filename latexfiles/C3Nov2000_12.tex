\JGWitem{Nov 2000, \#12}
For a 20-year term insurance on (70):
\begin{enumerate}
\item The death benefit, payable at the end of the year of death, is the net premium policy value plus 1000.
\item The one year mortality rate $q_{x+t}$ is equal to 0.03, for all $t\geq 0.$
\item $i = 0.07$
\end{enumerate}
Calculate the single benefit premium for this insurance.
\showMC{\begin{description}
\item[(A)] 216
\item[(B)] 267
\item[(C)] 318
\item[(D)] 369
\item[(E)] 420
\end{description}}

\showsol{\bsoln
\[
       \mbox{(Reserve at $n$) \ } +\mbox{ \ (Premium at $n$) \ } 
    =  vp_x\cdot\mbox{(Reserve at $n+1$)} \;+\; vq_x\cdot\mbox{(Death Ben. at $n+1$)} 
\]
\bears
  \prepostsub{n}{V}{} + \pi_n &=& vp_x (\prepostsub{n+1}{V}{}) + vq_{x+n}\bigl[1000 + (\prepostsub{n+1}{V}{})\bigr] \\
  (\prepostsub{n}{V}{}) - v(\prepostsub{n+1}{V}{}) + \pi_n &=& 1000vq_{x+n} \quad\mbox{ for each } \\ \\
  (n=0)\quad\quad\quad\quad - \underbrace{v(\prepostsub{1}{V}{})} + \pi_0 &=& 1000vq_{x} \\ \\
  (n=1)\quad\quad\prepostsub{1}{V}{} - v(\prepostsub{2}{V}{})  &=& 1000vq_{x+1} \\
                 \underbrace{v(\prepostsub{1}{V}{})} - v^2(\prepostsub{2}{V}{})  &=& 1000v^2q_{x+1} \\ 
  \mbox{(add eq'ns w/}\underbrace{\hspace{3pt}})\quad\quad \pi_0 - v^2(\prepostsub{2}{V}{})  &=& 1000(vq_x + v^2q_{x+1}) \\ \\
  (n=2)\quad\quad\prepostsub{2}{V}{} - v(\prepostsub{3}{V}{})  &=& 1000vq_{x+2} \\
                 v^2\prepostsub{2}{V}{} - v^3(\prepostsub{3}{V}{})  &=& 1000v^3q_{x+2} \\ 
  \mbox{(add eq'ns)}\quad\quad \pi_0 - v^3(\prepostsub{3}{V}{})  &=& 1000(vq_x + v^2q_{x+1}+v^3q_{x+2}) \\ \\
                         \vdots && \\ \\
  \pi_0 - v^{21}(\prepostsub{20}{V}{})  &=& 1000 \sum_{k=0}^{19}v^{k+1}q_{x+k} \\
  \pi_0 - 0  &=& 1000 \sum_{k=0}^{19}v^{k+1}q_{x+k} \\
  \pi_0  &=& 1000 \cdot(0.03) a_{\halfbox{20}} = 317.82
\eears

\esoln}
