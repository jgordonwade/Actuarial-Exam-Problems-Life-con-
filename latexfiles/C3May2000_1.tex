\JGWitem{May 2000, \#1}   % \S2 Single Life Survival
 Given:
\begin{enumerate}
 \item $\ecirc_0 = 25$, 
 \item $l_x = \omega - x$ for $0\leq x\leq \omega$, 
 \item $T(x)$ is the future lifetime random variable.
\end{enumerate}
Calculate $\Var{T(10)}$. 
\showMC{\begin{description}
\item (A) 65
\item (B) 93
\item (C) 133
\item (D) 178
\item (E) 333
\end{description}}
\showsol{\bsoln
 Since the survival function is linear, we know that 
$T(x)\sim\mbox{Uniform}(0,\omega-x)$, so that 
\bears
  \Expect{T(x)}&=& {b+a\over 2} = {\omega-x\over 2}, \\
  \Var{T(x)}&=& {(b-a)^2\over 12} = {(\omega-x)^2\over 12}.
\eears
Since $\Expect{T(0)}=\ecirc_0 = 25$, we know that $\Omega=50$, 
so $\Var{T(10)}= (50-10)^2/(12) = 133.\bar{3}$. 

\showMC{\underline{ANSWER: C}}

\esoln}
