\JGWitem{Course M, Nov 2006, \#39}
The random variable $N$ has a mixed distribution: 
\begin{enumerate}
\item With probability $p$, $N$ has a binomial distribution with $q = 0.5$ and $m= 2.$
\item With probability $1-p$, $N$ has a binomial distribution with $q = 0.5$ and $m = 4.$
\end{enumerate}
Which of the following is a correct expression for $\Prob(N = 2)$? 
\showMC{\begin{description}
\item[(A)] $0.125p^2$
\item[(B)] $0.375+0.125p$
\item[(C)] $0.375 + 0.125p^2$ 
\item[(D)] $0.375 - 0.125p^2$
\item[(E)] $0.375 - 0.125p$ 
\end{description}}

\leaveit{\bsoln
\bears
  \Prob{N=2} &=& \Prob{N=2|m=2}\Prob{m=2} \\
      && \;  + \; \Prob{N=2|m=4}\Prob{m=4} \\ 
    &=& p \left(\begin{array}{c}2 \\ 2\end{array}\right)(0.5)^{2}(0.5)^{2-2} 
       +  (1-p) \left(\begin{array}{c}4 \\ 2\end{array}\right)(0.5)^{2}(0.5)^{4-2} \\
    &=& \frac{1}{4} \cdot p + \frac{3}{8}(1-p) = \frac{3}{8} - \frac{1}{8}p 
\eears
\underline{Ans: E}

\esoln}
