\JGWitem{C3 May 2007, 04}
For a fully discrete whole life insurance of 150,000 on (x), you are given: 
\begin{enumerate}
\item $\pretwo{A}_x = 0.0143$
\item $A_x = 0.0653$
\item The annual premium is determined using the equivalence principle. 
\item $L$ is the loss-at-issue random variable. 
\end{enumerate}
Calculate the standard deviation of L. 
\showMC{\begin{description}
\item[(A)] 14.000 
\item[(B)] 14.500 
\item[(C)] 15.100 
\item[(D)] 15.600 
\item[(E)] 16.100 
\end{description}}

\showsol{\bsoln
\bears
 L &=& 150,000\times Z_x - \pi\times \ddot{Y}_x \\
   &=& 150,000\times Z_x - \frac{\pi}{d}\times (1-Z_x) \\
    \Expect{L}=0 &\Longrightarrow& \frac{\pi}{d} = 150,000\times\frac{A_x}{1-A_x} = 10,479.30 \\
 L &=& 150,000\times Z_x - 10,479.30\times \ddot{Y}_x \\
  &=& \left(150,000+10,479.30\right)\times Z_x + 10,479.30 \\
 \Var{L} &=& \left(150,000+10,479.30\right)^2 \left(\pretwo{A}_x-A_x^2\right)  = 2.584\,608\,69\times 10^8 \\
 \Std{L} &=& 16076.72
\eears
\esoln}
