\JGWitem{C3 Nov 2012 \#03}   % \S2 Single Life Survival
You are given:
\begin{enumerate}
\item $S_0(t) = \left(1-\frac{t}{\omega}\right)^{1/4}$ for $0\leq t \leq\omega$.
\item $\mu_{65} = \frac{1}{180}$
\end{enumerate}
Calculate the curtate expectation of life at age 106.

\showsol{\bsoln
We need to find $\omega$. Using the relation $\mu_{x} = -S'(x+t)/S(x)$, \\
we find that $\mu_{x} = \dfrac{1}{4(\omega-x)}$, so that 
$\mu_{65}=1/180$ implies $4(\omega-65) = 180$ or \underline{$\omega=110$.}

\bigskip

\bigskip
Now with $\omega=110$ we can proceed with the backward recursive relationship, $e_x = p_x(1+e_{x+1})$, where
\[
    p_x = \left(\dfrac{110-x-1}{110-x}\right)^{1/4}.
\]

\bears
   e_{110} = e_{109} &=& 0  \\
     p_{108} = (\dfrac{110-109}{108})^{1/4} =0.84090 \hspace{1.85in} e_{108} &=& 0.84090  \\
     p_{107}  = (\dfrac{110-108}{107})^{1/4} =0.90360 \quad\quad\quad e_{107} = 0.90360\cdot(1+0.84090) &=& 1.66344 \\
     p_{106}  = (\dfrac{110-107}{106})^{1/4} = 0.93061  \quad\quad\quad e_{106} = 0.9036\cdot(1+01.66344) &=& \boxed{2.47861 }
\eears

\esoln}
