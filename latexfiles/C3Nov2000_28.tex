\JGWitem{Nov 2000, \#28}
A decreasing term life insurance on (80) pays $(20-k)$ at the end of the year of death if $(80)$
dies in year $k+1,$ for $k=0,1,2,\ldots,19.$

You are given:
\begin{enumerate}
\item $i=0.06$
\item For a certain mortality table with $q_{80} = 0.2,$ the single benefit premium for this insurance is 13.
\item For this same mortality table, except that $q_{80} = 0.1,$ the single benefit premium for this insurance is P.
\end{enumerate}
Calculate $P$.

\showsol{\bsoln
\bears 
    13 &=& \sum_{k=0}^{19} (20-k)v^{k+1}\prepostsub{k|}{q}{80} \\
    &=& \frac{20}{1.06}\times (0.2) + \frac{v}{1.06}\times(1-0.2)\sum_{j=0}^{18} (19-j)\prepostsub{j|}{p}{81} \\
    13 &=& \frac{20}{1.06}\times (0.2) + \frac{1}{1.06}\times(1-0.2)\sum_{j=0}^{18} (19-j)\prepostsub{j|}{p}{81} \\  
    13 &=& \frac{(0.2)20}{1.06} + \frac{(0.8)}{1.06}\cdot X
\eears
where ${\displaystyle X=\sum_{j=0}^{18} (19-j)\prepostsub{j|}{p}{81}}$,
\[
    X = (13\cdot 1.06 - 4)/0.8 = 12.225
\]
Similarly, 
\bears
    P &=& \frac{(0.1)20}{1.06} + \frac{(0.9)}{1.06}\cdot X \\
    \boxed{P = 12.2665}
\eears

\esoln}
