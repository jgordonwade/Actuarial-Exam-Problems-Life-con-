\JGWitem{Nov 2008 \#13}   % \S2 Single Life Survival
You are given the curtate expectation of life at age x and x+1:
\[
   e_x = 5.000 \quad\quad e_{x+1} = 4.530.
\]
Calculate $q_x$.
%\showMC{\begin{description}
%\item[(A)] 
%\item[(B)] 
%\item[(C)] 
%\item[(D)] 
%\item[(E)] 
%\end{description}}
\showsol{\bsoln
  \bears
    e_x &=& p_x(1+e_{x+1}) \\
    q_x &=& 1-p_x-1 = \frac{e_x}{1+e_{x+1}} = 1-\frac{5.000}{4.530} = \underline{0.095841}
%    \showMC{\\\mbox{\sc Ans} &=& \mbox{\bf D}}
  \eears
\esoln}
