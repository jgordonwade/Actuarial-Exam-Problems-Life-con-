\JGWitem{ C3 Nov 2006, \# 33b}
Calculate the mean, the standard deviation, and the median of the random variable $\ax*{\angl{T_{40}}},$ 
assuming that $\omega = 120,$ that $\px[t]{40}=1-t/80,$ and that $\delta = 0.05.$

\showsol{\bsoln
The present value of a continuous annuity paying 1 per years for $t$ years is
\\ \dsy{\rule{0pt}{20pt}\ax*{\angl{t}}=\int_{z=0}^t e^{-\delta z}\,dz = \dfrac{1-e^{-\delta t}}{\delta} = 20\left(1-e^{-t/20}\right),}
{\ \ } so{ \ \ }
\underline{$  \ax*{\angl{T_{40}}} = 20\left(1-e^{T_{40}/20}\right).$}

\begin{description}
\item \underline{Mean}
   \[ \Expect{\ax*{\angl{T_{40}}}} = \ax*{40} = \int_0^{80} \left(20\left(1-e^{-t/20}\right)\right) \dfrac{1}{80}\,dt = \boxed{15.091578}. \]
\item \underline{Standard Deviation}
   \bears
     \Expect{\left(\ax*{\angl{T_{40}}}\right)^2}  &=& \int_0^{80} \left(20\left(1-e^{-t/20}\right)\right)^2 \dfrac{1}{80}\,dt = 25.890628, \\
     \Std{\ax*{\angl{T_{40}}}}  &=& \Expect{\left(\ax*{\angl{T_{40}}}\right)^2} - \left(\Expect{\ax*{\angl{T_{40}}}}\right)^2 = \boxed{5.088283}.
   \eears
\item \underline{Median}
The median of $Y$ is that value (call it $y_*$) 
for which \dsy{\rule{0pt}{20pt}\Prob{Y\leq y_*}=0.5.} So,
\bears
  0.5 = \Prob{Y\leq y_*} &=& \Prob{\dfrac{1-e^{-\delta T_{40}}}{\delta}\leq y_*}  \\
   &=& \Prob{ T_{40} \leq -\ln(1 - \delta y_*)/\delta } \\
   0.5 &=& \Prob{ T_{40} \leq t_* } \quad\text{ where }t_*= -\ln(1 - \delta y_*)/\delta . \\
   0.5 &=& 1-(1-t_*/80) = t_*/80 \\
   40 &=& t_* \quad\quad \boxed{y_* = 20\left(1-e^{-40/20}\right)  = 17.2933}
\eears

\end{description}
\esoln}
