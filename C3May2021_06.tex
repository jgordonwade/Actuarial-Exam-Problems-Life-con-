  \bears
      \ax**[30|]{30} &=& \dfrac{\ell_{60}}{\ell_{30}}v^{30}\ax**{60} \\
       &=& \dfrac{96634}{99338}\dfrac{1}{1.05^{30}}\cdot 14.90407 = \boxed{3.3546}
    \eears
\esoln}
