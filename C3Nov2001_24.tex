\JGWitem{Nov 2001, \#24}
 For a special 2-payment whole life insurance on (80):
\begin{enumerate}
\item Premiums of $\pi$ are paid at the beginning of years 1 and 3.
\item The death benefit is paid at the end of the year of death.
\item There is a partial refund of premium feature:
  \begin{description}
    \item If (80) dies in either year 1 or year 3, the death benefit is 
      $1000+\pi/2$
    \item Otherwise, the death benefit is $1000$.
  \end{description}
\item Mortality follows the Illustrative Life Table.
\item $i = 0.06$
\end{enumerate}
Calculate $\pi$, using the equivalence principle.
\showMC{\begin{description}
\item[(A)] 369
\item[(B)] 381
\item[(C)] 397
\item[(D)] 409
\item[(E)] 425
\end{description}}
\showsol{\bsoln
\bears
   \Expect{\text{Loss}}=0&=& \left(1000\Ax{80}+\dfrac{\pi}{2}v\qx{80}+\dfrac{\pi}{2}v^3(\qx[2|]{80})\right) 
      - \pi\left( 1+v^2\px[2]{80}\right) \\
  &=& 1000\Ax{80} - \pi\left(1+v^2\px[2]{80} - \dfrac{1}{2} v\qx{80}-\dfrac{1}{2}v^3\qx[2|]{80}\right) \\
  \pi &=& \dfrac{1000\Ax{80}}{1+v^2\px[2]{80} - \dfrac{1}{2} v\qx{80}-\dfrac{1}{2}v^3\qx[2|]{80}}
\eears
Now run the numbers.
\esoln}
