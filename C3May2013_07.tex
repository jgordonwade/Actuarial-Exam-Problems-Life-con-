\JGWitem{C3, May2013, \#07}
You are given:
\begin{enumerate}
\item $q_{60} = 0.01$
  \item Using $i=0.05,$ $A_{60:\halfbox{3}} = 0.86545$
\end{enumerate}
Using $i = 0.045$, calculate $A_{60:\halfbox{3}}.$
\showsol{\bsoln
  \bear
      A_{60:\halfbox{3}i} &=& \dfrac{q_{60}}{(1+i)} + \dfrac{\prepostsub{1|}{q}{60}}{(1+i)^2} + \dfrac{\prepostsub{2}{p}{60}}{(1+i)^3}  \nonumber  \\
       &=& \dfrac{0.01}{(1+i)} + \dfrac{0.99q_{61}}{(1+i)^2} + \dfrac{1-\bigl(0.01+(0.99q_{61}\bigr)}{(1+i)^3} \nonumber  \\
      A_{60:\halfbox{3}i} &=&  \left(\dfrac{0.01}{1+i} + \dfrac{0.99}{(1+i)^3}\right) + \left(\dfrac{0.99}{(1+i)^2}-\dfrac{0.99}{(1+i)^3}\right)q_{61}  
     \label{C3May2013_07_a}.
  \eear
   From the problem data we then find
   \bears
      A_{60:\halfbox{3}0.05} &=&  \left(\dfrac{0.01}{1.05} + \dfrac{0.99}{1.05^3}\right) + \left(\dfrac{0.99}{1.05^2}-\dfrac{0.99}{1.05^3}\right)q_{61}  \\
      0.86545  &=& 1.798056365 + 0.042759961\cdot q_{61}   \\
                     && \Longrightarrow\quad \underline{q_{61} = 0.017}
   \eears
   Using this value and $i=0.045$ in (\ref{C3May2013_07_a}),
   \bears
   A_{60:\halfbox{3}0.045} &=&  \left(\dfrac{0.01}{1.045} + \dfrac{0.99}{1.045^3}\right) + \left(\dfrac{0.99}{1.045^2}-\dfrac{0.99}{1.045^3}\right) \cdot 0.017 \\
   &=& \boxed{0.877766724}
  \eears

\esoln}
